Setup

makes here all the slow computations and cache them on disk

n_obs = 200
n_latents = range(1,4) 
data = GPFADataTest(hai[:n_obs])
cache_file_gaps = cache_path / "hai_train_multi_latents.pickle"
# cache_file_gaps.unlink() # uncomment this line to reset the cache
model_save_dir = here() / "analysis/trained_models"
imps = [GPFAImputation(data.data, latent_dims=i) for i in n_latents]
total_iter = 0
def train_save(n_iter=10):
    global total_iter
    total_iter += n_iter
    for imp, n_lat in zip(imps, n_latents):
        imp.fit(n_iter)
        imp.learner.save(model_save_dir / f"GPFA_l_{n_lat}_train_{total_iter}_1ker_{n_obs}_obs.pickle")
